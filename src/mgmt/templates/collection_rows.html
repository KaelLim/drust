              SSE quickstart (curl / Node / Browser)
              

                

                  curl
                  works with anon (needs select cap) or service token
                


curl -N \
  -H "Authorization: Bearer drust_service_…" \
  https://<your-drust-host>/drust/t/{{ tenant_id }}/records/{{ coll_name }}/subscribe


                

                  Node.js
                  native EventSource can't send headers — use the eventsource npm package
                


import EventSource from 'eventsource'; // npm i eventsource

const es = new EventSource(
  'https://<your-drust-host>/drust/t/{{ tenant_id }}/records/{{ coll_name }}/subscribe',
  { headers: { Authorization: 'Bearer drust_service_…' } }
);

es.addEventListener('created', e => console.log('+', JSON.parse(e.data).record));
es.addEventListener('updated', e => console.log('~', JSON.parse(e.data).record));
es.addEventListener('deleted', e => console.log('-', JSON.parse(e.data).id));


                

                  Browser
                  native EventSource can't send Authorization — proxy through a BFF, or use fetch + ReadableStream
                


// Anon tokens in URLs are unsafe — keep the bearer server-side and proxy the
// stream from your BFF. Direct fetch shown here for completeness.
const resp = await fetch(
  '/drust/t/{{ tenant_id }}/records/{{ coll_name }}/subscribe',
  { headers: { Authorization: 'Bearer drust_anon_…' } }
);
const reader = resp.body.getReader();
const decoder = new TextDecoder();
let buf = '';
for (;;) {
  const { value, done } = await reader.read();
  if (done) break;
  buf += decoder.decode(value, { stream: true });
  const frames = buf.split('\n\n'); buf = frames.pop();
  for (const frame of frames) {
    const evt = {};
    frame.split('\n').forEach(l => {
      const i = l.indexOf(':'); if (i < 0) return;
      evt[l.slice(0, i)] = l.slice(i + 1).trimStart();
    });
    console.log(evt.event, evt.data);
  }
}
